UPDATE nabory

    {{
      InputIlePrzekazanych.value === undefined ||

      InputIlePrzekazanych.value === ""

    NULLIF('{{ DatePickerProtokol.selectedDate ? moment(DatePickerProtokol.selectedDate).format("YYYY-MM-DD") : "" }}', '')::date,
    data_protokolu
  ),

  nr_uchwaly = COALESCE(NULLIF('{{ InputNrUchwaly.value || "" }}', ''), nr_uchwaly),

  data_uchwaly = COALESCE(
    NULLIF('{{ DatePickerUchwala.selectedDate ? moment(DatePickerUchwala.selectedDate).format("YYYY-MM-DD") : "" }}', '')::date,
    data_uchwaly

  status = COALESCE(NULLIF('{{ SelectStatus.value || "" }}', ''), status),

  stawka_panel = COALESCE(

      CurrencyOM.value === undefined ||
      CurrencyOM.value === null ||
      CurrencyOM.value === ""
        ? "NULL"
        : Number(CurrencyOM.value)
    }},
    stawka_om

WHERE id = {{
  TableNabory.selectedRow && TableNabory.selectedRow.id
