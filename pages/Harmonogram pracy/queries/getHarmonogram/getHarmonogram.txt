WITH params AS (
  SELECT
    {{ Number(SelectRok.selectedOptionValue || 2026) }}::int AS rok,
    {{ Number(SelectMiesiac.selectedOptionValue || 1) }}::int AS miesiac,
    EXTRACT(
      DAY FROM (
        date_trunc(
          'month',
          make_date(
            {{ Number(SelectRok.selectedOptionValue || 2026) }}::int,
            {{ Number(SelectMiesiac.selectedOptionValue || 1) }}::int,
            1
          )
        ) + interval '1 month - 1 day'
      )
    )::int AS dni_w_miesiacu
),
dane AS (
  SELECT
    h.id,
    h.pracownik_id,
    p.imie AS pracownik,
    h.rok,
    h.miesiac,
    h.dzien_1, h.dzien_2, h.dzien_3, h.dzien_4, h.dzien_5,
    h.dzien_6, h.dzien_7, h.dzien_8, h.dzien_9, h.dzien_10,
    h.dzien_11, h.dzien_12, h.dzien_13, h.dzien_14, h.dzien_15,
    h.dzien_16, h.dzien_17, h.dzien_18, h.dzien_19, h.dzien_20,
    h.dzien_21, h.dzien_22, h.dzien_23, h.dzien_24, h.dzien_25,
    h.dzien_26, h.dzien_27, h.dzien_28, h.dzien_29, h.dzien_30,
    h.dzien_31,
    h.uwagi
  FROM harmonogram_pracy h
  INNER JOIN pracownicy p
    ON p.id = h.pracownik_id
  INNER JOIN params par
    ON par.rok = h.rok
   AND par.miesiac = h.miesiac
  WHERE h.pracownik_id <> 1
),
wynik AS (

  SELECT
    0 AS sort_order,
    id AS "ID",
    pracownik AS "PRACOWNIK",
    dzien_1::text AS "01",
    dzien_2::text AS "02",
    dzien_3::text AS "03",
    dzien_4::text AS "04",
    dzien_5::text AS "05",
    dzien_6::text AS "06",
    dzien_7::text AS "07",
    dzien_8::text AS "08",
    dzien_9::text AS "09",
    dzien_10::text AS "10",
    dzien_11::text AS "11",
    dzien_12::text AS "12",
    dzien_13::text AS "13",
    dzien_14::text AS "14",
    dzien_15::text AS "15",
    dzien_16::text AS "16",
    dzien_17::text AS "17",
    dzien_18::text AS "18",
    dzien_19::text AS "19",
    dzien_20::text AS "20",
    dzien_21::text AS "21",
    dzien_22::text AS "22",
    dzien_23::text AS "23",
    dzien_24::text AS "24",
    dzien_25::text AS "25",
    dzien_26::text AS "26",
    dzien_27::text AS "27",
    dzien_28::text AS "28",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 29 THEN dzien_29::text ELSE NULL END AS "29",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 30 THEN dzien_30::text ELSE NULL END AS "30",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 31 THEN dzien_31::text ELSE NULL END AS "31",
    uwagi AS "UWAGI"
  FROM dane

  UNION ALL

  SELECT
    1 AS sort_order,
    NULL AS "ID",
    'W BIURZE' AS "PRACOWNIK",
    COUNT(*) FILTER (WHERE dzien_1 = 'B')::text AS "01",
    COUNT(*) FILTER (WHERE dzien_2 = 'B')::text AS "02",
    COUNT(*) FILTER (WHERE dzien_3 = 'B')::text AS "03",
    COUNT(*) FILTER (WHERE dzien_4 = 'B')::text AS "04",
    COUNT(*) FILTER (WHERE dzien_5 = 'B')::text AS "05",
    COUNT(*) FILTER (WHERE dzien_6 = 'B')::text AS "06",
    COUNT(*) FILTER (WHERE dzien_7 = 'B')::text AS "07",
    COUNT(*) FILTER (WHERE dzien_8 = 'B')::text AS "08",
    COUNT(*) FILTER (WHERE dzien_9 = 'B')::text AS "09",
    COUNT(*) FILTER (WHERE dzien_10 = 'B')::text AS "10",
    COUNT(*) FILTER (WHERE dzien_11 = 'B')::text AS "11",
    COUNT(*) FILTER (WHERE dzien_12 = 'B')::text AS "12",
    COUNT(*) FILTER (WHERE dzien_13 = 'B')::text AS "13",
    COUNT(*) FILTER (WHERE dzien_14 = 'B')::text AS "14",
    COUNT(*) FILTER (WHERE dzien_15 = 'B')::text AS "15",
    COUNT(*) FILTER (WHERE dzien_16 = 'B')::text AS "16",
    COUNT(*) FILTER (WHERE dzien_17 = 'B')::text AS "17",
    COUNT(*) FILTER (WHERE dzien_18 = 'B')::text AS "18",
    COUNT(*) FILTER (WHERE dzien_19 = 'B')::text AS "19",
    COUNT(*) FILTER (WHERE dzien_20 = 'B')::text AS "20",
    COUNT(*) FILTER (WHERE dzien_21 = 'B')::text AS "21",
    COUNT(*) FILTER (WHERE dzien_22 = 'B')::text AS "22",
    COUNT(*) FILTER (WHERE dzien_23 = 'B')::text AS "23",
    COUNT(*) FILTER (WHERE dzien_24 = 'B')::text AS "24",
    COUNT(*) FILTER (WHERE dzien_25 = 'B')::text AS "25",
    COUNT(*) FILTER (WHERE dzien_26 = 'B')::text AS "26",
    COUNT(*) FILTER (WHERE dzien_27 = 'B')::text AS "27",
    COUNT(*) FILTER (WHERE dzien_28 = 'B')::text AS "28",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 29 
      THEN COUNT(*) FILTER (WHERE dzien_29 = 'B')::text 
      ELSE NULL 
    END AS "29",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 30 
      THEN COUNT(*) FILTER (WHERE dzien_30 = 'B')::text 
      ELSE NULL 
    END AS "30",
    CASE WHEN (SELECT dni_w_miesiacu FROM params) >= 31 
      THEN COUNT(*) FILTER (WHERE dzien_31 = 'B')::text 
      ELSE NULL 
    END AS "31",
    NULL AS "UWAGI"
  FROM dane
)

SELECT *
FROM wynik
ORDER BY
  sort_order,
  "PRACOWNIK" COLLATE "pl-x-icu";
